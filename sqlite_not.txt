----DATA TYPES - VERÝ TÝPLERÝ----
int	- varsayýlan 11 haneli sayý üretilebilir.
varchar	- 255 karakterlik string alaný oluþturur
text	- uzun metinleri saklamak için
blob	- resim saklamak istiyorsak


----TABLO OLUÞTURMA----

Create Table Calisanlar(
	id int PRIMARY KEY,
	ad_soyad varchar,
	ise_giris_tarihi varchar,
	adresi text,
	telefon_no varchar
);
